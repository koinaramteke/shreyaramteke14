pipeline {
    agent any

    tools {
        maven 'Maven'   // Make sure Maven is configured in Jenkins Global Tools
    }

    stages {

        stage('PULL') {
            steps {
                git 'https://github.com/koinaramteke/shreyaramteke14.git'
                echo 'Pull Success'
            }
        }

        stage('BUILD') {
            steps {
                dir('backend') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('TEST') {
            steps {
                dir('backend') {
                    withSonarQubeEnv('sonarqube') {  
                        sh '''
                          mvn sonar:sonar \
                          -Dsonar.projectKey=student-add1 \
                          -Dsonar.projectName=student-add1 \
                        '''
                    }
                }
            }
        }

        stage('QUALITY GATE') {
            steps {
                timeout(time: 2, unit: 'MINUTES') {
                    waitForQualityGate abortPipeline: true
                }
            }
        }

        stage('DEPLOY') {
            steps {
                echo 'Deploy Success'
            }
        }
    }
}
